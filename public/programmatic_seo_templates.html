Complete script to generate all pages from CSV:

const fs = require('fs'); const csv = require('csv-parser'); const path = require('path'); // Load CSV and generate pages function generatePagesFromCSV(csvFile, templateType, outputDir) { const pages = []; fs.createReadStream(csvFile) .pipe(csv()) .on('data', (row) => { // Convert CSV row to template variables const variables = { NATURAL_QUERY: row.natural_query, SHORT_QUERY: row.short_query, EMOTIONAL_STATE: row.emotional_state, SITUATION: row.situation, MEDICAL_CONDITION: row.medical_condition, OPENING_SCENARIO: row.opening_scenario, VALIDATION_TEXT: row.validation_text, // Add Schema variables CRPO_NUMBER: "10979", THERAPIST_NAME: "Jesse Cynamon, RP" }; // Generate page HTML const pageHTML = generateSEOPage(templateType, variables); // CRPO Compliance Check if (checkCRPOCompliance(pageHTML)) { // Save file const filename = `${row.short_query}.html`; fs.writeFileSync(path.join(outputDir, filename), pageHTML); console.log(`✅ Generated: ${filename}`); } else { console.log(`❌ Failed compliance: ${row.natural_query}`); } }) .on('end', () => { console.log('✨ Page generation complete!'); }); } // CRPO Compliance Checker function checkCRPOCompliance(html) { const prohibitedTerms = [ 'best therapist', 'top rated', 'guaranteed', 'cure', 'testimonial', 'review', 'compared to', 'better than', 'only therapist' ]; const htmlLower = html.toLowerCase(); for (let term of prohibitedTerms) { if (htmlLower.includes(term)) { console.warn(`⚠️ Found prohibited term: "${term}"`); return false; } } // Check for required elements if (!html.includes('CRPO #10979')) { console.warn('⚠️ Missing CRPO registration number'); return false; } return true; } // Run the generator generatePagesFromCSV('./longtail-keywords.csv', 'longtail', './seo-pages/'); generatePagesFromCSV('./locations.csv', 'location', './location-pages/');
